CREATE TABLE ADMIN."DATE" 
    ( 
     DATEID   NUMBER GENERATED ALWAYS AS IDENTITY 
        ( START WITH 1 CACHE 20 ) , 
     FULLDATE DATE , 
     DAY      VARCHAR2 (10) , 
     MONTH    VARCHAR2 (10) , 
     YEAR     NUMBER , 
     QUARTER  VARCHAR2 (10) 
    ) 
    TABLESPACE DATA 
    LOGGING 
;


CREATE UNIQUE INDEX ADMIN.DATE_PK ON ADMIN."DATE" 
    ( 
     DATEID ASC 
    ) 
    TABLESPACE DATA 
    LOGGING 
;

ALTER TABLE ADMIN."DATE" 
    ADD CONSTRAINT DATE_PK PRIMARY KEY ( DATEID ) 
    USING INDEX ADMIN.DATE_PK ;

CREATE TABLE ADMIN.PRODUCT_DETAILS 
    ( 
     PRODUCTID            NUMBER GENERATED ALWAYS AS IDENTITY 
        ( START WITH 1 CACHE 20 ) , 
     PRODUCTNAME          VARCHAR2 (20) , 
     PRODUCTCATEGORY      VARCHAR2 (20) , 
     PRICEPERUNIT         NUMBER (8,2) , 
     AVAILABLEQTY         NUMBER , 
     PRODEXPIRYDATE       DATE , 
     PRODMANUFACTUREDDATE DATE 
    ) 
    TABLESPACE DATA 
    LOGGING 
;


CREATE UNIQUE INDEX ADMIN.PRODUCT_DETAILS_PK ON ADMIN.PRODUCT_DETAILS 
    ( 
     PRODUCTID ASC 
    ) 
    TABLESPACE DATA 
    LOGGING 
;

ALTER TABLE ADMIN.PRODUCT_DETAILS 
    ADD CONSTRAINT PRODUCT_DETAILS_PK PRIMARY KEY ( PRODUCTID ) 
    USING INDEX ADMIN.PRODUCT_DETAILS_PK ;

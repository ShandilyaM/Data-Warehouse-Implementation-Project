CREATE TABLE ADMIN.PAYMENT_DETAILS 
    ( 
     PAYMENTID     NUMBER GENERATED ALWAYS AS IDENTITY 
        ( START WITH 1 CACHE 20 ) , 
     PAYMENTMETHOD VARCHAR2 (10) , 
     PAYMENTAMOUNT NUMBER (8,2) , 
     PAYMENTDATE   TIMESTAMP 
    ) 
    TABLESPACE DATA 
    LOGGING 
;


CREATE UNIQUE INDEX ADMIN.PAYMENT_DETAILS_PK ON ADMIN.PAYMENT_DETAILS 
    ( 
     PAYMENTID ASC 
    ) 
    TABLESPACE DATA 
    LOGGING 
;

ALTER TABLE ADMIN.PAYMENT_DETAILS 
    ADD CONSTRAINT PAYMENT_DETAILS_PK PRIMARY KEY ( PAYMENTID ) 
    USING INDEX ADMIN.PAYMENT_DETAILS_PK ;
